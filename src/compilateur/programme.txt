public class Exemple_Grammaire {
// Autre méthode pour tester VarDecl et PrintStmt
    public static void testVariables() {
         a = 10;
        String texte = "Test";
        System.out.println(texte);
    }
    // Méthode avec paramètres String[] args
    public static void main(String[] args) {
        int x = 5;
        String message = "Bonjour";
       

        message = "Salut";
        System.out.println("message");

        if (x == 5) {
            System.out.println("x est égal à 5");
        }

        switch (x) {
            case 1:
                System.out.println("Premier cas");
                break;
            case 5:
                System.out.println("Deuxième cas");
                break;
            default:
                System.out.println("Cas par défaut");
        }

    }}
